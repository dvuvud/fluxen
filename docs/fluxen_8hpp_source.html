156inline constexpr uint8_t FLAG_LIVE = 0x00;

161/* On-disk entry header (6 bytes on disk) */
162struct EntryHeader {
163 uint8_t flags;
164 uint8_t key_len;
165 uint32_t val_len;
166};
167
168/* Serialize header into 6-byte buffer */
169inline void encode_header(uint8_t out[HEADER_SIZE],
170 const EntryHeader &h) noexcept {
171 out[0] = h.flags;
172 out[1] = h.key_len;
173 out[2] = static_cast<uint8_t>(h.val_len & 0xFFu);
174 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
175 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
176 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
177}
178
179/* Deserialise header from a 6-byte buffer */
180inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept
181 -> EntryHeader {
182 return {
183 .flags = in[0],
184 .key_len = in[1],
185 .val_len = static_cast<uint32_t>(in[2]) |
186 static_cast<uint32_t>(in[3]) << 8 |
187 static_cast<uint32_t>(in[4]) << 16 |
188 static_cast<uint32_t>(in[5]) << 24,
189 };
190}

1064 void transaction(const std::function<TxResult(Tx &)> &fn) {
1065 check_poisoned();

1067 TxResult result = fn(tx);
1068 if (result == rollback) {
1069 return;
1070 }
1071
1072 std::vector<uint8_t> batch;
1073 batch.reserve(tx.ops_.size() * 64);
1074 for (const auto &op : tx.ops_) {
1075 detail::EntryHeader hdr{
1076 .flags = op.is_delete ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1077 .key_len = static_cast<uint8_t>(op.key.size()),
1078 .val_len = op.is_delete ? 0u : static_cast<uint32_t>(op.val.size()),
1079 };
1080 uint8_t raw[detail::HEADER_SIZE];
1081 detail::encode_header(raw, hdr);
1082 batch.insert(batch.end(), raw, raw + detail::HEADER_SIZE);
1083 batch.insert(batch.end(), op.key.begin(), op.key.end());
1084 if (!op.is_delete) {
1085 batch.insert(batch.end(), op.val.begin(), op.val.end());
1086 }
1087 }
1088
1089 std::unique_lock lock(mu_);
1090
1091 const size_t size_before = file_.size();
1092
1093 if (!file_.append(batch.data(), batch.size())) {
1094 throw io_error("fluxen: transaction append failed");
1095 }
1096
1097 if (!file_.sync()) {
1098 if (!file_.truncate(size_before)) {
1099 poisoned_ = true;
1100 throw poisoned_error(
1101 "fluxen: transaction fsync failed and truncation failed. Database "
1102 "file may contain a partial tail entry");
1103 }
1104 throw io_error("fluxen: transaction fsync failed");
1105 }
1106
1107 size_t pos = 0;
1108 for (const auto &op : tx.ops_) {
1109 pos += detail::HEADER_SIZE + op.key.size();
1110 if (op.is_delete) {
1111 index_.erase(op.key);
1112 } else {
1113 index_[op.key] = {.val_offset = size_before + pos,
1114 .val_len = static_cast<uint32_t>(op.val.size())};
1115 pos += op.val.size();
1116 }
1117 }

1250 void load_index() {

1252 throw corrupt_error("fluxen: file too small to be valid");
1253 }
1254

1256 throw corrupt_error(

1258 }

1261 size_t last_good_pos = pos;
1262
1263 while (pos + detail::HEADER_SIZE <= file_.size()) {
1264 uint8_t raw[detail::HEADER_SIZE];

1266 detail::EntryHeader hdr = detail::decode_header(raw);
1267 pos += detail::HEADER_SIZE;

1270 break;
1271 }

1274 hdr.key_len);
1275 pos += hdr.key_len;
1276
1277 if (hdr.flags == detail::FLAG_TOMB) {
1278 index_.erase(key);
1279 } else {
1280 index_[key] = {.val_offset = pos, .val_len = hdr.val_len};

1285 }

1332 }

1334 }

1338 index_.erase(it);
1339 }
1340 } else {

1342 .val_len = val_len};
1343 }
1344 }
