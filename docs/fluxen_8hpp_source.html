139/* On-disk entry header (6 bytes on disk) */
140struct EntryHeader {
141 uint8_t flags;
142 uint8_t key_len;
143 uint32_t val_len;
144};
145
146/* Serialize header into 6-byte buffer */
147inline void encode_header(uint8_t out[HEADER_SIZE],
148 const EntryHeader &h) noexcept {
149 out[0] = h.flags;
150 out[1] = h.key_len;
151 out[2] = static_cast<uint8_t>(h.val_len & 0xFFu);
152 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
153 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
154 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
155}
156
157/* Deserialise header from a 6-byte buffer */
158inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept
159 -> EntryHeader {
160 return {
161 .flags = in[0],
162 .key_len = in[1],
163 .val_len = static_cast<uint32_t>(in[2]) |
164 static_cast<uint32_t>(in[3]) << 8 |
165 static_cast<uint32_t>(in[4]) << 16 |
166 static_cast<uint32_t>(in[5]) << 24,
167 };
168}

986 void transaction(const std::function<TxResult(Tx &)> &fn) {
987 check_poisoned();
988 Tx tx;
989 TxResult result = fn(tx);
990 if (result == rollback) {
991 return;
992 }
993
994 std::vector<uint8_t> batch;
995 batch.reserve(tx.ops_.size() * 64);
996 for (const auto &op : tx.ops_) {
997 detail::EntryHeader hdr{
998 .flags = op.is_delete ? detail::FLAG_TOMB : detail::FLAG_LIVE,
999 .key_len = static_cast<uint8_t>(op.key.size()),
1000 .val_len = op.is_delete ? 0u : static_cast<uint32_t>(op.val.size()),
1001 };
1002 uint8_t raw[detail::HEADER_SIZE];
1003 detail::encode_header(raw, hdr);
1004 batch.insert(batch.end(), raw, raw + detail::HEADER_SIZE);
1005 batch.insert(batch.end(), op.key.begin(), op.key.end());
1006 if (!op.is_delete) {
1007 batch.insert(batch.end(), op.val.begin(), op.val.end());
1008 }
1009 }
1010
1011 std::unique_lock lock(mu_);
1012
1013 const size_t size_before = file_.size();
1014
1015 if (!file_.append(batch.data(), batch.size())) {
1016 throw std::runtime_error("fluxen: transaction append failed");
1017 }
1018
1019 if (!file_.sync()) {
1020 if (!file_.truncate(size_before)) {
1021 poisoned_ = true;
1022 throw std::runtime_error(
1023 "fluxen: transaction fsync failed and truncation failed. Database file may contain a partial tail entry");
1024 }
1025 throw std::runtime_error("fluxen: transaction fsync failed");
1026 }

1029 for (const auto &op : tx.ops_) {
1030 pos += detail::HEADER_SIZE + op.key.size();

1033 } else {
1034 index_[op.key] = {.val_offset = size_before + pos,
1035 .val_len = static_cast<uint32_t>(op.val.size())};
1036 pos += op.val.size();
1037 }
1038 }

1214 void append_entry(std::string_view key, const uint8_t *val, uint32_t val_len,
1215 bool tombstone) {
1216 assert(key.size() <= detail::MAX_KEY &&
1217 "fluxen: key exceeds 255-byte limit");
1218
1219 detail::EntryHeader hdr{
1220 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
1221 .key_len = static_cast<uint8_t>(key.size()),
1222 .val_len = val_len,
1223 };
1224
1225 std::vector<uint8_t> buf;
1226 buf.resize(detail::HEADER_SIZE + key.size() + val_len);
1227 detail::encode_header(buf.data(), hdr);
1228 std::memcpy(buf.data() + detail::HEADER_SIZE, key.data(), key.size());
1229 if (val && val_len) {
1230 std::memcpy(buf.data() + detail::HEADER_SIZE + key.size(), val, val_len);
1231 }
1232
1233 const size_t size_before = file_.size();
1234
1235 if (!file_.append(buf.data(), buf.size())) {
1236 if (!file_.truncate(size_before)) {
1237 poisoned_ = true;
1238 throw std::runtime_error(
1239 "fluxen: append failed and truncation failed. Database file may contain a partial tail entry");
1240 }
1241 throw std::runtime_error("fluxen: append failed");
1242 }
1243
1244 if (tombstone) {
1245 if (auto it = index_.find(key); it != index_.end()) {
1246 index_.erase(it);
1247 }
1248 } else {
1249 index_[std::string(key)] = {.val_offset = file_.size() - val_len,
1250 .val_len = val_len};
1251 }
