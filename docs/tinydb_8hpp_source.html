121/* On-disk entry header (6 bytes on disk) */
122struct EntryHeader {
123 uint8_t flags;
124 uint8_t key_len;
125 uint32_t val_len;
126};
127
128/* Serialize header into 6-byte buffer */
129inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
130 out[0] = h.flags;
131 out[1] = h.key_len;
132 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
133 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
134 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
135 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
136}
137
138/* Deserialise header from a 6-byte buffer */
139inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept -> EntryHeader {
140 return {
141 .flags = in[0],
142 .key_len = in[1],
143 .val_len = static_cast<uint32_t>(in[2])
144 | static_cast<uint32_t>(in[3]) << 8
145 | static_cast<uint32_t>(in[4]) << 16
146 | static_cast<uint32_t>(in[5]) << 24,
147 };
148}

199 auto remap() -> bool {
200 unmap();
201 size_ = file_size();
202 if (size_ == 0) {
203 return true;
204 }
205
206#ifdef _WIN32
207 map_ = CreateFileMappingA(file_, nullptr, PAGE_READWRITE, 0, 0, nullptr);
208
209 if (!map_) {
210 return false;
211 }
212
213 ptr_ = static_cast<uint8_t*>(MapViewOfFile(map_, FILE_MAP_ALL_ACCESS, 0, 0, size_));
214
215 if (!ptr_) {
216 CloseHandle(map_);
217 map_ = nullptr;
218 return false;
219 }
220#else
221 ptr_ = static_cast<uint8_t*>(::mmap(nullptr, size_, PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0));
222
223 if (ptr_ == MAP_FAILED) {
224 ptr_ = nullptr;
225 return false;
226 }
227#endif
228 return true;
229 }

231 auto append(const void* data, size_t len) -> bool {
232 if (len == 0) {
233 return true;

235#ifdef _WIN32

238 return WriteFile(file_, data, static_cast<DWORD>(len), &written, nullptr)
239 && written == static_cast<DWORD>(len);
240#else

242#endif

253 auto rewrite(const std::vector<uint8_t>& data) -> bool {
254 unmap();
255#ifdef _WIN32
256 SetFilePointer(file_, 0, nullptr, FILE_BEGIN);
257 SetEndOfFile(file_);
258 DWORD written = 0;
259 if (!WriteFile(file_, data.data(), static_cast<DWORD>(data.size()),
260 &written, nullptr)) {
261 return false;
262 }
263 FlushFileBuffers(file_);
264#else
265 if (::ftruncate(fd_, 0) < 0) return false;
266 ::lseek(fd_, 0, SEEK_SET);
267 if (::write(fd_, data.data(), data.size())
268 != static_cast<ssize_t>(data.size()))
269 return false;
270 ::fsync(fd_);
271#endif
272 return remap();
273 }

538 template <typename T = std::string>
539 auto get(std::string_view key) const -> std::optional<T> {
540 std::scoped_lock lock(mu_);
541 auto it = index_.find(std::string(key));
542 if (it == index_.end()) {
543 return std::nullopt;
544 }
545
546 const auto& entry = it->second;
547
548 if constexpr (std::is_same_v<T, std::string>) {
549 const auto* ptr = file_.ptr() + entry.val_offset;
550 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
551 } else {
552 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");
553 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
554 return std::nullopt;
555 }
556 T result;
557 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
558 return result;
559 }

704 void transaction(const std::function<TxResult(Tx&)>& fn) {
705 Tx tx;
706 TxResult result = fn(tx);
707 if (result == rollback) return;
708
709 std::scoped_lock lock(mu_);
710 for (const auto& op : tx.ops_) {
711 append_entry(
712 op.key,
713 op.is_delete ? nullptr : op.val.data(),
714 op.is_delete ? 0 : op.val.size(),
715 op.is_delete
716 );
717 }
718 file_.sync();
719 }

748 void compact() {
749 std::scoped_lock lock(mu_);
750
751 std::vector<uint8_t> buf;
752 buf.reserve(file_.size());
753
754 buf.insert(
755 buf.end(),
756 detail::MAGIC,
757 detail::MAGIC + sizeof(detail::MAGIC)
758 );
759
760 std::unordered_map<std::string, detail::IndexEntry> new_index;
761 for (const auto& [key, entry] : index_) {
762 detail::EntryHeader hdr{
763 .flags = detail::FLAG_LIVE,
764 .key_len = static_cast<uint8_t>(key.size()),
765 .val_len = entry.val_len,
766 };
767
768 uint8_t raw[detail::HEADER_SIZE];
769 detail::encode_header(raw, hdr);
770
771 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
772
773 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
774 buf.insert(buf.end(), key.begin(), key.end());
775
776 const auto* val_ptr = file_.ptr() + entry.val_offset;
777 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
778
779 new_index[key] = { .val_offset=val_off, .val_len=entry.val_len };
780 }
781
782 if (!file_.rewrite(buf)) {
783 throw std::runtime_error("tinydb: compact failed during rewrite");
784 }
785
786 index_ = std::move(new_index);
787 }

830 void load_index() {
831 if (file_.size() < sizeof(detail::MAGIC)) {
832 throw std::runtime_error("tinydb: file too small to be valid");
833 }
834
835 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
836 throw std::runtime_error("tinydb: bad magic. File was not created by tinydb");
837 }
838
839 size_t pos = sizeof(detail::MAGIC);
840
841 while (pos + detail::HEADER_SIZE <= file_.size()) {
842 uint8_t raw[detail::HEADER_SIZE];
843 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
844 detail::EntryHeader hdr = detail::decode_header(raw);
845 pos += detail::HEADER_SIZE;
846
847 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
848 break;
849 }
850
851 std::string key(
852 reinterpret_cast<const char*>(file_.ptr() + pos),
853 hdr.key_len
854 );
855 pos += hdr.key_len;
856
857 if (hdr.flags == detail::FLAG_TOMB) {
858 index_.erase(key);
859 } else {
860 index_[key] = { .val_offset=pos, .val_len=hdr.val_len };
861 }
862
863 pos += hdr.val_len;
864 }
865 }

876 void append_entry(std::string_view key,
877 const uint8_t* val,
878 uint32_t val_len,
879 bool tombstone) {
880 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");
881
882 detail::EntryHeader hdr{
883 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
884 .key_len = static_cast<uint8_t>(key.size()),
885 .val_len = val_len,
886 };
887
888 uint8_t raw[detail::HEADER_SIZE];
889 detail::encode_header(raw, hdr);
890
891 file_.append(raw, detail::HEADER_SIZE);
892 file_.append(key.data(), key.size());
893 if (val && val_len) {
894 file_.append(val, val_len);
895 }
896 file_.remap();
897
898 if (tombstone) {
899 index_.erase(std::string(key));
900 } else {
901 size_t val_off = file_.size() - val_len;
902 index_[std::string(key)] = { .val_offset=val_off, .val_len=val_len };
903 }
904 }
