123/* On-disk entry header (6 bytes on disk) */
124struct EntryHeader {
125 uint8_t flags;
126 uint8_t key_len;
127 uint32_t val_len;
128};
129
130/* Serialize header into 6-byte buffer */
131inline void encode_header(uint8_t out[HEADER_SIZE], const EntryHeader& h) noexcept {
132 out[0] = h.flags;
133 out[1] = h.key_len;
134 out[2] = static_cast<uint8_t>( h.val_len & 0xFFu);
135 out[3] = static_cast<uint8_t>((h.val_len >> 8) & 0xFFu);
136 out[4] = static_cast<uint8_t>((h.val_len >> 16) & 0xFFu);
137 out[5] = static_cast<uint8_t>((h.val_len >> 24) & 0xFFu);
138}
139
140/* Deserialise header from a 6-byte buffer */
141inline auto decode_header(const uint8_t in[HEADER_SIZE]) noexcept -> EntryHeader {
142 return {
143 .flags = in[0],
144 .key_len = in[1],
145 .val_len = static_cast<uint32_t>(in[2])
146 | static_cast<uint32_t>(in[3]) << 8
147 | static_cast<uint32_t>(in[4]) << 16
148 | static_cast<uint32_t>(in[5]) << 24,
149 };
150}

201 auto remap() -> bool {
202 unmap();
203 size_ = file_size();
204 dirty_ = false;
205 file_size_ = size_;
206 if (size_ == 0) {
207 return true;
208 }
209
210#ifdef _WIN32

212
213 if (!map_) {
214 return false;
215 }

217 ptr_ = static_cast<uint8_t*>(MapViewOfFile(map_, FILE_MAP_ALL_ACCESS, 0, 0, size_));
218
219 if (!ptr_) {
220 CloseHandle(map_);
221 map_ = nullptr;
222 return false;
223 }
224#else

227 if (ptr_ == MAP_FAILED) {
228 ptr_ = nullptr;
229 return false;
230 }
231#endif

235 auto append(const void* data, size_t len) -> bool {
236 if (len == 0) {
237 return true;
238 }
239#ifdef _WIN32
240 DWORD written = 0;
241 SetFilePointer(file_, 0, nullptr, FILE_END);
242 if (!WriteFile(file_, data, static_cast<DWORD>(len), &written, nullptr)
243 || written != static_cast<DWORD>(len)) {
244 return false;
245 }
246#else
247 ::lseek(fd_, 0, SEEK_END);
248 if (::write(fd_, data, len) != static_cast<ssize_t>(len)) {
249 return false;
250 }
251#endif
252 dirty_ = true;
253 file_size_ += len;
254 return true;

266 unmap();

269 SetEndOfFile(file_);
270 DWORD written = 0;
271 if (!WriteFile(file_, data.data(), static_cast<DWORD>(data.size()),
272 &written, nullptr)) {
273 return false;
274 }
275 FlushFileBuffers(file_);
276#else
277 if (::ftruncate(fd_, 0) < 0) return false;
278 ::lseek(fd_, 0, SEEK_SET);
279 if (::write(fd_, data.data(), data.size())
280 != static_cast<ssize_t>(data.size()))
281 return false;
282 ::fsync(fd_);
283#endif
284 return remap();
285 }

564 template <typename T = std::string>
565 auto get(std::string_view key) const -> std::optional<T> {
566 std::scoped_lock lock(mu_);
567 auto it = index_.find(std::string(key));
568 if (it == index_.end()) {
569 return std::nullopt;
570 }
571
572 const auto& entry = it->second;
573
574 if constexpr (std::is_same_v<T, std::string>) {
575 const auto* ptr = file_.ptr() + entry.val_offset;
576 return std::string(reinterpret_cast<const char*>(ptr), entry.val_len);
577 } else {
578 static_assert(std::is_trivially_copyable_v<T>, "tinydb: T must be trivially copyable");
579 if (entry.val_len != static_cast<uint32_t>(sizeof(T))) {
580 return std::nullopt;
581 }
582 T result;
583 std::memcpy(&result, file_.ptr() + entry.val_offset, sizeof(T));
584 return result;
585 }

730 void transaction(const std::function<TxResult(Tx&)>& fn) {
731 Tx tx;
732 TxResult result = fn(tx);
733 if (result == rollback) return;
734
735 std::scoped_lock lock(mu_);
736 for (const auto& op : tx.ops_) {
737 append_entry(
738 op.key,
739 op.is_delete ? nullptr : op.val.data(),
740 op.is_delete ? 0 : op.val.size(),
741 op.is_delete
742 );
743 }
744 file_.sync();

774 void compact() {
775 std::scoped_lock lock(mu_);
776
777 std::vector<uint8_t> buf;
778 buf.reserve(file_.size());
779
780 buf.insert(
781 buf.end(),
782 detail::MAGIC,
783 detail::MAGIC + sizeof(detail::MAGIC)
784 );
785
786 std::unordered_map<std::string, detail::IndexEntry> new_index;
787 for (const auto& [key, entry] : index_) {
788 detail::EntryHeader hdr{
789 .flags = detail::FLAG_LIVE,
790 .key_len = static_cast<uint8_t>(key.size()),
791 .val_len = entry.val_len,
792 };
793
794 uint8_t raw[detail::HEADER_SIZE];
795 detail::encode_header(raw, hdr);
796
797 size_t val_off = buf.size() + detail::HEADER_SIZE + key.size();
798
799 buf.insert(buf.end(), raw, raw + detail::HEADER_SIZE);
800 buf.insert(buf.end(), key.begin(), key.end());
801
802 const auto* val_ptr = file_.ptr() + entry.val_offset;
803 buf.insert(buf.end(), val_ptr, val_ptr + entry.val_len);
804
805 new_index[key] = { .val_offset=val_off, .val_len=entry.val_len };
806 }
807
808 if (!file_.rewrite(buf)) {
809 throw std::runtime_error("tinydb: compact failed during rewrite");
810 }
811
812 index_ = std::move(new_index);
813 }

855 void load_index() {
856 if (file_.size() < sizeof(detail::MAGIC)) {
857 throw std::runtime_error("tinydb: file too small to be valid");
858 }
859
860 if (std::memcmp(file_.ptr(), detail::MAGIC, sizeof(detail::MAGIC)) != 0) {
861 throw std::runtime_error("tinydb: bad magic. File was not created by tinydb");
862 }
863
864 size_t pos = sizeof(detail::MAGIC);
865
866 while (pos + detail::HEADER_SIZE <= file_.size()) {
867 uint8_t raw[detail::HEADER_SIZE];
868 std::memcpy(raw, file_.ptr() + pos, detail::HEADER_SIZE);
869 detail::EntryHeader hdr = detail::decode_header(raw);
870 pos += detail::HEADER_SIZE;
871
872 if (pos + hdr.key_len + hdr.val_len > file_.size()) {
873 break;
874 }
875
876 std::string key(
877 reinterpret_cast<const char*>(file_.ptr() + pos),
878 hdr.key_len
879 );
880 pos += hdr.key_len;
881
882 if (hdr.flags == detail::FLAG_TOMB) {
883 index_.erase(key);
884 } else {
885 index_[key] = { .val_offset=pos, .val_len=hdr.val_len };
886 }
887
888 pos += hdr.val_len;
889 }

901 void append_entry(std::string_view key,
902 const uint8_t* val,
903 uint32_t val_len,
904 bool tombstone) {
905 assert(key.size() <= detail::MAX_KEY && "tinydb: key exceeds 255-byte limit");
906
907 detail::EntryHeader hdr{
908 .flags = tombstone ? detail::FLAG_TOMB : detail::FLAG_LIVE,
909 .key_len = static_cast<uint8_t>(key.size()),
910 .val_len = val_len,
911 };
912
913 uint8_t raw[detail::HEADER_SIZE];
914 detail::encode_header(raw, hdr);
915
916 file_.append(raw, detail::HEADER_SIZE);
917 file_.append(key.data(), key.size());
918 if (val && val_len) {
919 file_.append(val, val_len);
920 }
921
922 if (tombstone) {
923 index_.erase(std::string(key));
924 } else {
925 size_t val_off = file_.size() - val_len;
926 index_[std::string(key)] = { .val_offset=val_off, .val_len=val_len };
927 }
928 }
